		w = 0;
		T f = 1;
		char ch = getchar();
		while (ch < '0' || ch > '9') {
			if (ch == '-')
				f = -1;
			ch = getchar();
		}
		while (ch <= '9' && ch >= '0') {
			w = w * 10 + ch - '0';
			ch = getchar();
		}
	}

		if (x < 0) {

		putchar(x % 10 + '0');
	}
